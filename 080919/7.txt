#!/bin/bash
# Task No 7
a=5
b=7
# echo file{$a..$b} - не работает, потому что сначала обрабатывается brace expansion, а уже потом variable expansion. Поэтому нужно экранировать все кроме переменных, чтобы они выполнились сначала, и запусть выполнение по второму кругу с помощью команды eval:
eval 'echo file{'$a'..'$b'}'
# так как в данном случае без заменённых переменных выражение в скобках не валидно для замены, то можно опустить ковычки и просто выполнить второй раз:
eval echo file{$a..$b}
