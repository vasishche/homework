#!/bin/bash
# Task No 3
tail -n 2 /etc/fstab
head -n 7 /etc/yum.conf
# число строк в файле
numberOfStrings=$(cat /etc/fstab | wc -l)
# если в командах tail и head после -n указать больше строк, чем есть в файле, то просто будет выведен весь файл
# на 1 больше чем число строк в файле
echo $((numberOfStrings+1))
# столько же сколько и число строк в файле
tail -n $((numberOfStrings+1)) /etc/fstab | wc -l
head -n $((numberOfStrings+1)) /etc/fstab | wc -l
